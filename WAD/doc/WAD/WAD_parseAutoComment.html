0022 function WAD_parseAutoComment( sSeries, sAutoComment )
0023 % - parse the <autoComment> field in configuration file
0024 % - syntax:
0025 %     <autoComment>
0026 %         <description>Scanner Software Version</description>
0027 %         <field>SoftwareVersion<field>
0028 %       <level>1<level>
0029 %     </autoComment>
0030 % Takes the value from DICOM field SoftwareVersion and prints it as result
0031 % in the defined level.
0032 %
0033 % ------------------------------------------------------------------------
0034 % WAD
0035 % file: WAD_parseAutoComment
0036 %
0037 % ------------------------------------------------------------------------
0038 % VUmc, Amsterdam, NL / Joost Kuijer / jpa.kuijer@vumc.nl
0039 % 2013-09-06 / JK
0040 % V1.0: new function
0041 % ------------------------------------------------------------------------
0042 
0043 
0044 % ----------------------
0045 % GLOBALS
0046 % ----------------------
0047 global WAD
0048 
0049 
0050 % version info
0051 my.name = 'WAD_parseAutoComment';
0052 my.version = '1.0';
0053 my.date = '20130906';
0054 WAD_vbprint( ['Module ' my.name ' Version ' my.version ' (' my.date ')'], 2 );
0055 
0056 
0057 % sanity check
0058 if isempty( sAutoComment )
0059     return
0060 end
0061 
0062 if WAD.versionmodus > 1
0063     % WAD 2 style config
0064     % syntax is: ac.description='<your description>'; ac.field='<DICOM field name>'; ac.level=<level>;
0065     try
0066         commentToEvaluate = sAutoComment;
0067         eval( commentToEvaluate );
0068         sAutoComment = ac;
0069     catch err
0070         disp( ['ERROR in ' my.name ' while evaluating autoComment string.'] );
0071         disp( ['"' commentToEvaluate '"'] );
0072         disp( err.message );
0073         return
0074     end
0075 end
0076 
0077         
0078         
0079 % number of comments in list
0080 nComments = length( sAutoComment );
0081 
0082 % read the DICOM image
0083 ci = 1; % current instance: first image
0084 WAD_vbprint( [my.name ':   loading DICOM image file: ' sSeries.instance(ci).filename ] );
0085 try
0086     dcmInfo = dicominfo( sSeries.instance(ci).filename );
0087 catch err
0088     WAD_ErrorMsg( my.name, ['ERROR reading DICOM info from file "' sSeries.instance(ci).filename '")'], err );
0089     return
0090 end
0091 
0092 
0093 % loop over comments
0094 for i_ic = 1:nComments
0095     theDescription = []; % actually a valid description
0096     if isfield( sAutoComment(i_ic), 'description' )
0097         theDescription = sAutoComment(i_ic).description;
0098     end
0099     
0100     theLevel = 2; % default output to level 2 results
0101     if isfield( sAutoComment(i_ic), 'level' ) && ~isempty( sAutoComment(i_ic).level ) && isnumeric( sAutoComment(i_ic).level )
0102         theLevel = sAutoComment(i_ic).level;
0103     end
0104     
0105     if isfield( sAutoComment(i_ic), 'field' ) && ~isempty( sAutoComment(i_ic).field )
0106         theField = sAutoComment(i_ic).field;
0107         if isfield( dcmInfo, theField )
0108             theComment = [];
0109             
0110             if isstruct( dcmInfo.(theField) )
0111                 % can't process structured data fields
0112             elseif ischar( dcmInfo.(theField) )
0113                 theComment = dcmInfo.(theField);
0114             elseif isnumeric( dcmInfo.(theField) )
0115                 theComment = num2str( dcmInfo.(theField) );
0116             end
0117             
0118             % do not export empty comment fields
0119             if isempty( theComment )
0120                 WAD_vbprint( [my.name ': Error: field content of ' theField ' is not String or Numeric format, or DICOM field is empty.'] );
0121             else
0122                 % output comment to results file
0123                 WAD_resultsAppendString( theLevel, theComment, theDescription );
0124             end
0125             
0126         else
0127             % DICOM field not present
0128             WAD_vbprint( [my.name ': Error: no field named ' theField ' in DICOM header.'] );
0129         end
0130     end
0131 end
