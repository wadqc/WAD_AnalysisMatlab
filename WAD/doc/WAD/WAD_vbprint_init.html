0022 function WAD_vbprint_init( logfilename )
0023 % vbprint is used for tracing / debugging / logging
0024 % where the text output goes to, is defined by WAD.cfg.logverbose
0025 % init: check XML fields and open log file
0026 
0027 global WAD
0028 
0029 default.mode = 1;
0030 default.level = 1;
0031 
0032 % check logverbose configuration fields
0033 if ~isfield( WAD.cfg, 'logverbose' )
0034     WAD_resultsAppendString( 2, 'Field logverbose missing in configuration file. Using default settings.', 'WARNING' );
0035     % apply default settings
0036     WAD.cfg.logverbose = default;
0037 end
0038 
0039 if ~isfield( WAD.cfg.logverbose, 'mode' )
0040     WAD_resultsAppendString( 2, 'Field logverbose.mode missing in configuration file. Using default settings.', 'WARNING' );
0041     % apply default settings
0042     WAD.cfg.logverbose.mode = default.mode;
0043 end
0044 if ~isfield( WAD.cfg.logverbose, 'level' )
0045     WAD_resultsAppendString( 2, 'Field logverbose.level missing in configuration file. Using default settings.', 'WARNING' );
0046     % apply default settings
0047     WAD.cfg.logverbose.level = default.level;
0048 end
0049 
0050 
0051 % open log file
0052 WAD.cfg.logverbose.fid = -1;
0053 WAD.cfg.logverbose.fullfile = fullfile( WAD.in.analysemodule_outputdir, logfilename );
0054 WAD.cfg.logverbose.fid = fopen( WAD.cfg.logverbose.fullfile, 'w' );
0055 if WAD.cfg.logverbose.fid < 0
0056     % error opening the file
0057     WAD_resultsAppendString( 2, ['Could not open log file "' WAD.cfg.logverbose.fullfile '"'], 'ERROR' );
0058     return
0059 end
0060 
0061 return
