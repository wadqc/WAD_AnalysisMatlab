0022 function WAD_vbprint_init( logfilename )
0023 % vbprint is used for tracing / debugging / logging
0024 % where the text output goes to, is defined by WAD.cfg.logverbose
0025 % init: check XML fields and open log file
0026 
0027 global WAD
0028 
0029 default.mode = 1;
0030 default.level = 1;
0031 
0032 % WAD2 compatibility
0033 if WAD.versionmodus > 1
0034     try
0035         WAD.cfg.logverbose.mode = str2double( WAD.cfg.actions.logverbose.params.logMode );
0036         WAD.cfg.logverbose.level = str2double( WAD.cfg.actions.logverbose.params.logLevel );
0037     catch err
0038         disp( 'WARNING: Error interpreting configured logverbose settings. Using default settings.' );
0039         disp( err.message );
0040     end
0041 end
0042 
0043 % check logverbose configuration fields
0044 if ~isfield( WAD.cfg, 'logverbose' )
0045     disp( 'WARNING: Field logverbose missing in configuration file. Using default settings.' );
0046     % apply default settings
0047     WAD.cfg.logverbose = default;
0048 end
0049 
0050 if ~isfield( WAD.cfg.logverbose, 'mode' )
0051     disp( 'WARNING: Field logverbose.mode missing in configuration file. Using default settings.' );
0052     % apply default settings
0053     WAD.cfg.logverbose.mode = default.mode;
0054 end
0055 if ~isfield( WAD.cfg.logverbose, 'level' )
0056     disp( 'WARNING: Field logverbose.level missing in configuration file. Using default settings.' );
0057     % apply default settings
0058     WAD.cfg.logverbose.level = default.level;
0059 end
0060 
0061 
0062 % open log file
0063 WAD.cfg.logverbose.fid = -1;
0064 WAD.cfg.logverbose.fullfile = fullfile( WAD.in.analysemodule_outputdir, logfilename );
0065 WAD.cfg.logverbose.fid = fopen( WAD.cfg.logverbose.fullfile, 'w' );
0066 if WAD.cfg.logverbose.fid < 0
0067     % error opening the file
0068     disp( ['ERROR: Could not open log file "' WAD.cfg.logverbose.fullfile '"'] );
0069     return
0070 end
0071 
0072 return
