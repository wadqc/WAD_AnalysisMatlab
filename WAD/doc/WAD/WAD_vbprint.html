0022 function WAD_vbprint( arg, level )
0023 % vbprint is used for tracing / debugging / logging
0024 % where the text output goes to, is defined by WAD.cfg.logverbose
0025 
0026 global WAD
0027 
0028 
0029 if ( nargin ~= 2 )
0030     level = 1;
0031 end
0032 
0033 if ( level <= WAD.cfg.logverbose.level )
0034     if ( bitand( WAD.cfg.logverbose.mode, 1 ) && ( WAD.cfg.logverbose.fid ~= -1 ) )
0035         fprintf( WAD.cfg.logverbose.fid, '%s\n', arg );
0036     end
0037 
0038     if ( bitand( WAD.cfg.logverbose.mode, 2 ) )
0039         % print to screen
0040         fprintf( '%s\n', arg );
0041     end  
0042 end
0043 
0044 return
