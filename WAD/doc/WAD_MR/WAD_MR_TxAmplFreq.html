0167 % get requested number from the dicom header
0168 if isfield( fieldinfo, 'type' )
0169     if strcmp( fieldinfo.type, 'char' )
0170         % convert this dicom field to the requested type 'char'
0171         x = cast( dicomheader.(fieldinfo.field)(:), fieldinfo.type )';
0172

0179                 %numend = strfind( x(numstart:end), char(fieldinfo.pattern(2)) );
0180                 numend = strfind( x(numstart:end), char(10) );
0181                 if numend
0182                     if ( numend < 3 )
0183                         WAD_vbprint( [my.name ': Error: reading from header (no number between starting and trailing patterns).'] );
0184                         content = -1;
0185                     end
0186                     content = str2num( x(numstart:numstart+numend(1)-2) );
0187                 else
0188                     WAD_vbprint( [my.name ': Error: reading from header (trailing pattern not found).'] );
0189                     content = -1;
0190                 end
0191             else
0192                 WAD_vbprint( [my.name ': Error: reading from header (starting pattern not found).'] );
0193                 content = -1;
0194             end
0195         else
0196             % no pattern to search for, just convert...
0197             content = str2num( x );
0198         end % pattern
0199     else
0200         % not char, we need to typecast the field
0201         content = double( typecast( dicomheader.(fieldinfo.field), fieldinfo.type ) );
0202     end
0203 else
0204     % no additional field type specifiers... should be a number known in
0205     % dicom dictionary
