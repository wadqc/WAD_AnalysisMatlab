0144 % Local function that reads the numerical value from a DICOM field.
0145 % Implemented are

0148 function content = getField( dicomheader, fieldinfo )
0149 my.name = 'WAD_MR_TxAmplFreq:getField';
0150 % get requested number from the dicom header
0151 if isfield( fieldinfo, 'type' )
0152     if strcmp( fieldinfo.type, 'char' )
0153         % convert this dicom field to the requested type 'char'
0154         x = cast( dicomheader.(fieldinfo.field)(:), fieldinfo.type )';
0155 
0156         if isfield( fieldinfo, 'pattern' )
0157             % find the starting pattern
0158             %numstart = strfind( x, char(fieldinfo.pattern(1)) ) + length( char(fieldinfo.pattern(1)) );
0159             numstart = strfind( x, fieldinfo.pattern ) + length( fieldinfo.pattern );
0160             if numstart
0161                 % find the trailing pattern
0162                 %numend = strfind( x(numstart:end), char(fieldinfo.pattern(2)) );
0163                 numend = strfind( x(numstart:end), char(10) );
0164                 if numend
0165                     if ( numend < 3 )
0166                         WAD_vbprint( [my.name ': Error: reading from header (no number between starting and trailing patterns).'] );
0167                         content = -1;
0168                     end
0169                     content = str2num( x(numstart:numstart+numend(1)-2) );
0170                 else
0171                     WAD_vbprint( [my.name ': Error: reading from header (trailing pattern not found).'] );
0172                     content = -1;
0173                 end
0174             else
0175                 WAD_vbprint( [my.name ': Error: reading from header (starting pattern not found).'] );
0176                 content = -1;
0177             end
0178         else
0179             % no pattern to search for, just convert...
0180             content = str2num( x );
0181         end % pattern
0182     else
0183         % not char, we need to typecast the field
0184         content = double( typecast( dicomheader.(fieldinfo.field), fieldinfo.type ) );
0185     end
0186 else
0187     % no additional field type specifiers... should be a number known in
0188     % dicom dictionary
